import { createServerClient } from '@supabase/ssr'
import { cookies } from 'next/headers'

export function createClient() {
  const cookieStore = cookies()
  return createServerClient(
    process.env.NEXT_PUBLIC_SUPABASE_URL!,
    process.env.NEXT_PUBLIC_SUPABASE_ANON_KEY!,
    {
      cookies: {
        getAll() {
          return cookieStore.getAll()
        },
        setAll(cookiesToSet) {
          try {
            cookiesToSet.forEach(({ name, value, options }) =>
              cookieStore.set(name, value, options)
            )
          } catch {
            // setAll can be called from Server Components where cookies are read-only
          }
        },
      },
    }
  )
}

**Step 4: Rewrite middleware.ts** with RBAC role guard (per RBAC-03, RBAC-04).

Use the new `getAll/setAll` cookie API. For auth validation, call `supabase.auth.getUser()` which validates the JWT against the Supabase Auth server. For reading the `user_role` custom claim, read the raw access token cookie directly from the request cookies and decode it with `jose` `decodeJwt`. Do NOT call `getSession()` — it is an anti-pattern on the server per RESEARCH.md.

CRITICAL — Borrower route protection: The `ROLE_ROUTES` map MUST include `/dashboard` (and all other borrower portal URL prefixes) mapped to `'borrower'`. The borrower portal pages live under `app/(portal)/` which maps to URLs `/dashboard`, `/payments`, `/invoices`, `/statements`, `/contact`. Without this, an admin or note_buyer who is authenticated could access borrower pages freely.

```typescript
import { createServerClient } from '@supabase/ssr'
import { NextResponse, type NextRequest } from 'next/server'
import { decodeJwt } from 'jose'

// Which role is required for each route prefix
const ROLE_ROUTES: Record<string, string> = {
  '/admin': 'admin',
  '/buyer': 'note_buyer',
  '/dashboard': 'borrower',
  '/payments': 'borrower',
  '/invoices': 'borrower',
  '/statements': 'borrower',
  '/contact': 'borrower',
}

// Where each role's "home" is (for wrong-role redirects per user decision)
const ROLE_REDIRECT: Record<string, string> = {
  admin: '/admin',
  note_buyer: '/buyer',
  borrower: '/dashboard',
}

export async function middleware(request: NextRequest) {
  let supabaseResponse = NextResponse.next({ request })

  const supabase = createServerClient(
    process.env.NEXT_PUBLIC_SUPABASE_URL!,
    process.env.NEXT_PUBLIC_SUPABASE_ANON_KEY!,
    {
      cookies: {
        getAll() {
          return request.cookies.getAll()
        },
        setAll(cookiesToSet) {
          cookiesToSet.forEach(({ name, value }) =>
            request.cookies.set(name, value)
          )
          supabaseResponse = NextResponse.next({ request })
          cookiesToSet.forEach(({ name, value, options }) =>
            supabaseResponse.cookies.set(name, value, options)
          )
        },
      },
    }
  )

  // Validate auth — getUser() verifies JWT with Auth server (NOT getSession)
  const { data: { user } } = await supabase.auth.getUser()

  const { pathname } = request.nextUrl

  // Unauthenticated — redirect to login (preserve intended destination)
  if (!user) {
    const loginUrl = request.nextUrl.clone()
    loginUrl.pathname = '/login'
    loginUrl.searchParams.set('redirect', pathname)
    return NextResponse.redirect(loginUrl)
  }

  // Read user_role from JWT custom claim (injected by custom_access_token_hook)
  // No DB call — the role is embedded in the token
  // Read the raw access token directly from the cookie — NOT from getSession()
  let userRole: string | null = null
  const accessTokenCookie = request.cookies.getAll().find(c =>
    c.name.endsWith('-auth-token') ||
    c.name.includes('sb-') && c.name.includes('-auth-token')
  )
  if (accessTokenCookie) {
    try {
      // The Supabase auth cookie may be base64-encoded or chunked.
      // For a simple single-chunk cookie, the value is the raw JWT.
      const payload = decodeJwt(accessTokenCookie.value)
      userRole = (payload as Record<string, unknown>).user_role as string | null
    } catch {
      // If cookie decode fails, try reading individual chunk cookies
      // Supabase SSR stores tokens as sb-<ref>-auth-token.0, .1, etc.
      const allCookies = request.cookies.getAll()
      const tokenChunks = allCookies
        .filter(c => c.name.includes('-auth-token'))
        .sort((a, b) => a.name.localeCompare(b.name))
      if (tokenChunks.length > 0) {
        try {
          const combined = tokenChunks.map(c => c.value).join('')
          // The combined value may be base64url-encoded JSON containing the access_token
          const parsed = JSON.parse(
            Buffer.from(combined.replace('base64-', ''), 'base64url').toString()
          )
          if (parsed.access_token) {
            const payload = decodeJwt(parsed.access_token)
            userRole = (payload as Record<string, unknown>).user_role as string | null
          }
        } catch {
          userRole = null
        }
      }
    }
  }

  // Enforce role per route prefix
  for (const [routePrefix, requiredRole] of Object.entries(ROLE_ROUTES)) {
    if (pathname.startsWith(routePrefix) && userRole !== requiredRole) {
      // Redirect to user's own portal — not a generic error page (per user decision)
      const ownPortal = userRole ? ROLE_REDIRECT[userRole] ?? '/login' : '/login'
      const redirectUrl = request.nextUrl.clone()
      redirectUrl.pathname = ownPortal
      // Copy cookies from supabaseResponse to preserve refreshed session
      const redirectResponse = NextResponse.redirect(redirectUrl)
      supabaseResponse.cookies.getAll().forEach(cookie => {
        redirectResponse.cookies.set(cookie)
      })
      return redirectResponse
    }
  }

  return supabaseResponse
}

export const config = {
  matcher: [
    '/admin/:path*',
    '/buyer/:path*',
    '/dashboard/:path*',
    '/payments/:path*',
    '/invoices/:path*',
    '/statements/:path*',
    '/contact/:path*',
    '/((?!_next/static|_next/image|favicon.ico|login|auth|.*\\.(?:svg|png|jpg|jpeg|gif|webp)$).*)',
  ],
}
```

IMPORTANT: The middleware must return `supabaseResponse` (not a new NextResponse.next()) so refreshed session cookies propagate. When redirecting, copy cookies from supabaseResponse onto the redirect response.

IMPORTANT: `decodeJwt` from `jose` does NOT verify the signature — it only decodes. This is acceptable here because `getUser()` already validated authentication. We only need to read the claim value from the already-validated session.

NOTE: If `getClaims()` is available on the installed `@supabase/ssr` version, the executor MAY use it instead of the raw cookie decode approach. `getClaims()` validates the JWT signature locally and returns the decoded payload including custom claims — it is the cleanest approach. Check: `const { data: { claims } } = await supabase.auth.getClaims()`. If it exists, use it and remove the manual cookie/decode logic.
